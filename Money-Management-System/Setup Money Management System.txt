CREATE DATABASE money_manager;
USE money_manager;
CREATE TABLE balance (id INT NOT NULL AUTO_INCREMENT, uid INT NOT NULL, amount DOUBLE(10,2) NOT NULL, last_dt DATETIME NOT NULL, PRIMARY KEY (id,uid));
CREATE TABLE record (id INT NOT NULL AUTO_INCREMENT, uid INT NOT NULL, type VARCHAR(20) NOT NULL, amount DOUBLE(10,2) NOT NULL, purpose VARCHAR(100), dt DATETIME NOT NULL, PRIMARY KEY (id,uid));
COMMIT;
